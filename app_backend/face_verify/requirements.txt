# Face engine: dlib 128-D via face_recognition (same as the face-attendance app).
# The engine runs as a child process of the Node backend (face_worker.py over
# stdin/stdout) — there is NO HTTP service, so fastapi/uvicorn are NOT needed.
#
# dlib is a native build — on Windows install a prebuilt wheel if `pip install dlib`
# fails (e.g. `pip install dlib-bin`), then `pip install face_recognition`.
# On Linux: apt-get install -y build-essential cmake, then pip install.
#
# Install on a server:  bash setup_engine.sh  (creates face_verify/venv from this file).
# opencv-python-HEADLESS: the plain opencv-python wheel needs libGL/X11 system libs
# that a headless server lacks ("ImportError: libGL.so.1"), so use the headless build
# — it provides the same cv2 module the worker imports, with no GUI deps.
face_recognition>=1.3.0
dlib>=19.24.0
opencv-python-headless>=4.8.0
numpy>=1.24.0
